


FILE(GLOB OPENFLUID_CORE_CPP *.cpp)
FILE(GLOB OPENFLUID_CORE_HPP *.hpp)


ADD_LIBRARY(openfluid-core SHARED ${OPENFLUID_CORE_CPP} ${OPENFLUID_CORE_HPP})

SET_TARGET_PROPERTIES(openfluid-core PROPERTIES VERSION "${VERSION_MAJOR}.${VERSION_MINOR}" 
                                        SOVERSION "${VERSION_MAJOR}.${VERSION_MINOR}.${VERSION_PATCH}"
                                        LIBRARY_OUTPUT_DIRECTORY "${LIB_OUTPUT_PATH}"
                                        RUNTIME_OUTPUT_DIRECTORY "${LIB_OUTPUT_PATH}"
                                        ARCHIVE_OUTPUT_DIRECTORY "${LIB_OUTPUT_PATH}")



INSTALL(TARGETS openfluid-core
        RUNTIME DESTINATION ${BIN_INSTALL_PATH}
        LIBRARY DESTINATION ${LIB_INSTALL_PATH}
        ARCHIVE DESTINATION ${LIB_INSTALL_PATH})
   

INSTALL(FILES ${OPENFLUID_CORE_HPP} DESTINATION ${INCLUDE_INSTALL_PATH}/core)
