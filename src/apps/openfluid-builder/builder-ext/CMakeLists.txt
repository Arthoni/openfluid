
INCLUDE_DIRECTORIES(.)

FILE(GLOB OPENFLUID_BUILDEREXT_CPP *.cpp)
FILE(GLOB OPENFLUID_BUILDEREXT_HPP *.hpp)


ADD_LIBRARY(openfluid-builder-ext SHARED ${OPENFLUID_BUILDEREXT_CPP} ${OPENFLUID_BUILDEREXT_HPP})

SET_TARGET_PROPERTIES(openfluid-builder-ext PROPERTIES VERSION "${VERSION_MAJOR}.${VERSION_MINOR}" 
                                        SOVERSION "${VERSION_MAJOR}.${VERSION_MINOR}.${VERSION_PATCH}"
                                        LIBRARY_OUTPUT_DIRECTORY "${LIB_OUTPUT_PATH}"
                                        RUNTIME_OUTPUT_DIRECTORY "${LIB_OUTPUT_PATH}"
                                        ARCHIVE_OUTPUT_DIRECTORY "${LIB_OUTPUT_PATH}")

INSTALL(TARGETS openfluid-builder-ext
        RUNTIME DESTINATION ${BIN_INSTALL_PATH}
        LIBRARY DESTINATION ${LIB_INSTALL_PATH}
        ARCHIVE DESTINATION ${LIB_INSTALL_PATH})
   

INSTALL(FILES ${OPENFLUID_BUILDEREXT_HPP} DESTINATION ${INCLUDE_INSTALL_PATH}/builderext)
